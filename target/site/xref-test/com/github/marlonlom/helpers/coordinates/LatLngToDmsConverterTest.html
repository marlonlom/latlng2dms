19    @Test
20    public final void shouldFailConvertingLatitude93negative() {

22        final Double value = -93d;

24        Assert.fail("Latitude converted to DMS: " + converted);
25      } catch (Exception e) {
26        final String expectedMsg = "Latitude must not be greater then 90 degrees";
27        Assert.assertEquals(expectedMsg, e.getMessage());
28      }
29    }
